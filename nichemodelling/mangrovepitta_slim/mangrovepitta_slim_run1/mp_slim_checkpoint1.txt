// mangrove pitta simulation with no genetics
initialize() {
	// define some global constants for parameters
	
	// define mutation rate
	defineConstant("MU", 4.6e-9);
	
	// define number of mutating sites
	defineConstant("L", 10000);
	
	// define recombination rate
	defineConstant("R", 1e-9);
	
	//define population size
	defineConstant("N", 10000);
	
	// define adult death rate
	defineConstant("D", 0.263507866);

	// set model type as non-Wright-Fisher
	initializeSLiMModelType("nonWF");

	// set up model as a spatial simulation
	initializeSLiMOptions(dimensionality="xy");

	//set up sexual model
	initializeSex("A");

	// carrying-capacity density per sqkm
	defineConstant("K", 40); 

	// SIGMA_S, the spatial competition width
	defineConstant("S", 0.1); 

	// spatial competition
	initializeInteractionType(1, "xy", reciprocal=T, maxDistance=S * 3);
	i1.setInteractionFunction("n", 1.0, S);
	// spatial mate choice
	initializeInteractionType(2, "xy", reciprocal=T, maxDistance=0.5);
	i2.setConstraints("receiver", sex="F", minAge=1);
	i2.setConstraints("exerter", sex="M", minAge=1);

	// starting time interval for map choice (depends on BPP results)
	defineConstant("startInt", 514);
	
	// checkpoint resumption value
	defineConstant("chkpoint", 1);

	//mutation rate 
	initializeMutationRate(MU);
	
	// m1 mutation type: neutral
	initializeMutationType("m1", 0.5, "f", 0.0);
	m1.convertToSubstitution = T;
	
	// g1 genomic element type: uses m1 for all mutations
	initializeGenomicElementType("g1", m1, 1.0);
	
	// uniform chromosome of length 100 kb with uniform recombination
	initializeGenomicElement(g1, 0, L - 1);
	initializeRecombinationRate(R);
}

modifyChild() {
	do {
		do pos = parent1.spatialPosition + rnorm(2, 0, 8.89);
		while (!p1.pointInBounds(pos));
	} while (p1.spatialMapValue("SEA",pos) == 0.0);

	child.setSpatialPosition(pos);
	return T;
}
